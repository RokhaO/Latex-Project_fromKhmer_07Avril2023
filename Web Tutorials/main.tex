	\section{Introduction}
	\begin{frame}{What is HTML?}
		HTML is the standard markup language for creating Web pages.
		\begin{itemize}
			 \item<1-> HTML stands for Hyper Text Markup Language.
			 \item<2-> HTML describes the structure of Web pages using markup
			 \item<3-> HTML elements are the building blocks of HTML pages
			 \item<4-> HTML elements are represented by tags
			 \item <5-> HTML tags label pieces of content such as "heading", "paragraph", "table", and so on
			 \item <6-> Browsers do not display the HTML tags, but use them to render the content of the page
		\end{itemize}
	\end{frame}
	\subsection{A Simple HTML Document}
		\begin{lstlisting}

			<html>

			<body>
			
			<h1>My First Heading</h1>
			<p>My first paragraph.</p>
			
			</body>
			</html>
